# F0.26 GAN生成对抗网络

## 算法概述

生成对抗网络（GAN）由生成器和判别器组成，通过对抗训练学习数据分布。在电池管理系统中，GAN主要用于：

- **数据增强**：生成合成电池数据
- **故障模拟**：生成故障样本
- **图像生成**：极片缺陷图像生成
- **风格迁移**：不同电池类型数据转换

## 目标函数

```
min_G max_D V(D, G) = E_{x~p_data}[log D(x)] + E_{z~p_z}[log(1 - D(G(z)))]

其中：
- D: 判别器（区分真假）
- G: 生成器（生成假数据）
```

## 代码示例

```python
import torch
import torch.nn as nn

class BatteryGAN(nn.Module):
    """
    电池数据生成GAN

    生成器：噪声 → 电池时序数据
    判别器：判断数据真假
    """
    def __init__(self, latent_dim, data_dim):
        super().__init__()

        # 生成器
        self.generator = nn.Sequential(
            nn.Linear(latent_dim, 128),
            nn.LeakyReLU(0.2),
            nn.Linear(128, 256),
            nn.LeakyReLU(0.2),
            nn.Linear(256, data_dim),
            nn.Tanh()  # 输出范围[-1, 1]
        )

        # 判别器
        self.discriminator = nn.Sequential(
            nn.Linear(data_dim, 256),
            nn.LeakyReLU(0.2),
            nn.Linear(256, 128),
            nn.LeakyReLU(0.2),
            nn.Linear(128, 1),
            nn.Sigmoid()  # 输出概率[0, 1]
        )

    def generate(self, z):
        return self.generator(z)

    def discriminate(self, x):
        return self.discriminator(x)

# 训练GAN
def train_gan(dataloader, latent_dim, data_dim, n_epochs=100):
    """
    训练GAN生成电池数据
    """
    gan = BatteryGAN(latent_dim, data_dim)
    optimizer_g = torch.optim.Adam(gan.generator.parameters(), lr=0.0002)
    optimizer_d = torch.optim.Adam(gan.discriminator.parameters(), lr=0.0002)
    criterion = nn.BCELoss()

    for epoch in range(n_epochs):
        for real_data in dataloader:
            # 训练判别器
            optimizer_d.zero_grad()

            # 真实数据
            real_labels = torch.ones(real_data.size(0), 1)
            real_loss = criterion(gan.discriminate(real_data), real_labels)

            # 生成数据
            z = torch.randn(real_data.size(0), latent_dim)
            fake_data = gan.generate(z)
            fake_labels = torch.zeros(real_data.size(0), 1)
            fake_loss = criterion(gan.discriminate(fake_data.detach()), fake_labels)

            d_loss = real_loss + fake_loss
            d_loss.backward()
            optimizer_d.step()

            # 训练生成器
            optimizer_g.zero_grad()

            z = torch.randn(real_data.size(0), latent_dim)
            fake_data = gan.generate(z)
            g_loss = criterion(gan.discriminate(fake_data), real_labels)

            g_loss.backward()
            optimizer_g.step()

    return gan
```
