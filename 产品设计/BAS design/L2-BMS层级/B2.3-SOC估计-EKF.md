## 算法原理

### Thevenin等效电路模型

```
状态变量: X = [SOC, V_p1, V_p2]^T

状态方程:
  SOC(k+1) = SOC(k) - I(k)·Δt/Cn
  V_p1(k+1) = V_p1(k)·exp(-Δt/τ1) + I(k)·R1·(1-exp(-Δt/τ1))
  V_p2(k+1) = V_p2(k)·exp(-Δt/τ2) + I(k)·R2·(1-exp(-Δt/τ2))

观测方程:
  V(k) = OCV(SOC(k)) - I(k)·R0 - V_p1(k) - V_p2(k)
```

### EKF递推

```python
def ekf_predict(x, P, I, dt):
    """预测步骤"""
    # 状态转移
    x_pred = A @ x + B * I

    # 协方差预测
    P_pred = A @ P @ A.T + Q

    return x_pred, P_pred

def ekf_update(x_pred, P_pred, V_meas, I):
    """更新步骤"""
    # 观测预测
    V_pred = h(x_pred, I)

    # 新息
    y_tilde = V_meas - V_pred

    # 卡尔曼增益
    K = P_pred @ H.T @ inv(H @ P_pred @ H.T + R)

    # 状态更新
    x_new = x_pred + K * y_tilde

    # 协方差更新
    P_new = (I - K @ H) @ P_pred

    return x_new, P_new
```

## 代码示例（Python验证版）

```python
import numpy as np

class EKF_SOC:
    def __init__(self, Cn, R0, R1, R2, C1, C2, Q, R):
        self.Cn = Cn  # 额定容量
        self.R0 = R0  # 欧姆内阻
        self.R1, self.R2 = R1, R2
        self.C1, self.C2 = C1, C2
        self.tau1, self.tau2 = R1*C1, R2*C2

        # 状态: [SOC, Vp1, Vp2]
        self.x = np.array([0.5, 0.0, 0.0])
        self.P = np.eye(3) * 0.1

        # 噪声协方差
        self.Q = Q  # 过程噪声
        self.R = R  # 观测噪声

    def ocv_func(self, soc):
        """OCV-SOC关系（简化为线性，实际用查表）"""
        return 3.2 + 0.4 * soc  # LFP简化模型

    def predict(self, I, dt):
        """预测步骤"""
        # 状态转移矩阵A
        A = np.array([
            [1, 0, 0],
            [0, np.exp(-dt/self.tau1), 0],
            [0, 0, np.exp(-dt/self.tau2)]
        ])

        # 输入矩阵B
        B = np.array([
            [-dt/self.Cn],
            [self.R1 * (1 - np.exp(-dt/self.tau1))],
            [self.R2 * (1 - np.exp(-dt/self.tau2))]
        ])

        # 状态预测
        self.x = A @ self.x + B.flatten() * I

        # 协方差预测
        self.P = A @ self.P @ A.T + self.Q

    def update(self, V_meas, I):
        """更新步骤"""
        # 观测预测
        V_pred = self.ocv_func(self.x[0]) - I*self.R0 - self.x[1] - self.x[2]

        # 观测矩阵H（雅可比矩阵）
        H = np.array([
            [0.4, -1, -1]  # d(OCV)/d(SOC) ≈ 0.4 (LFP)
        ])

        # 新息
        y_tilde = V_meas - V_pred

        # 卡尔曼增益
        S = H @ self.P @ H.T + self.R
        K = self.P @ H.T / S

        # 状态更新
        self.x = self.x + K.flatten() * y_tilde

        # 协方差更新
        self.P = (np.eye(3) - np.outer(K, H)) @ self.P

    def get_soc(self):
        return self.x[0]
```
