# F0.32 PPO近端策略优化

## 算法概述

PPO是一种策略梯度算法，通过信任域优化提高稳定性。在电池管理系统中，PPO主要用于：

- **充电电流控制**：连续电流值控制
- **充电策略优化**：多目标优化
- **热管理**：温度控制策略

## 目标函数

```
L^{CLIP}(θ) = E[min(r_t(θ)·Â_t, clip(r_t(θ), 1-ε, 1+ε)·Â_t)]

其中：
- r_t(θ) = π_θ(a_t|s_t) / π_{θ_old}(a_t|s_t)（重要性采样）
- Â_t: 优势估计
- ε: 裁剪参数（通常为0.2）
```

## 代码示例

```python
import torch
import torch.nn as nn
import torch.optim as optim
from torch.distributions import Normal

class PPOActor(nn.Module):
    """
    PPO Actor网络（策略网络）
    """
    def __init__(self, state_dim, action_dim):
        super().__init__()

        self.network = nn.Sequential(
            nn.Linear(state_dim, 64),
            nn.Tanh(),
            nn.Linear(64, 64),
            nn.Tanh()
        )

        self.mu_head = nn.Linear(64, action_dim)  # 均值
        self.log_std_head = nn.Linear(64, action_dim)  # 标准差

    def forward(self, x):
        x = self.network(x)
        mu = self.mu_head(x)
        log_std = self.log_std_head(x)
        std = torch.exp(log_std)
        return mu, std

class PPOCritic(nn.Module):
    """
    PPO Critic网络（价值网络）
    """
    def __init__(self, state_dim):
        super().__init__()

        self.network = nn.Sequential(
            nn.Linear(state_dim, 64),
            nn.Tanh(),
            nn.Linear(64, 64),
            nn.Tanh(),
            nn.Linear(64, 1)
        )

    def forward(self, x):
        return self.network(x)

class PPOAgent:
    """
    PPO智能体
    """
    def __init__(self, state_dim, action_dim, lr=3e-4):
        self.actor = PPOActor(state_dim, action_dim)
        self.critic = PPOCritic(state_dim)

        self.actor_optimizer = optim.Adam(self.actor.parameters(), lr=lr)
        self.critic_optimizer = optim.Adam(self.critic.parameters(), lr=lr)

        # PPO参数
        self.epsilon = 0.2  # 裁剪参数
        self.gamma = 0.99  # 折扣因子
        self.gae_lambda = 0.95  # GAE参数
        self.clip_epochs = 10  # PPO epochs
        self.batch_size = 64

    def select_action(self, state):
        """
        选择动作
        """
        with torch.no_grad():
            state = torch.FloatTensor(state).unsqueeze(0)
            mu, std = self.actor(state)
            dist = Normal(mu, std)
            action = dist.sample()
            log_prob = dist.log_prob(action)

        return action[0].numpy(), log_prob[0].numpy()

    def compute_gae(self, rewards, values, dones):
        """
        计算广义优势估计（GAE）
        """
        advantages = []
        gae = 0

        values = values + [0]  # 添加最后一个值

        for t in reversed(range(len(rewards))):
            delta = rewards[t] + self.gamma * values[t + 1] * (1 - dones[t]) - values[t]
            gae = delta + self.gamma * self.gae_lambda * (1 - dones[t]) * gae
            advantages.insert(0, gae)

        return advantages

    def update(self, states, actions, old_log_probs, rewards, dones):
        """
        更新策略
        """
        # 计算values
        states = torch.FloatTensor(states)
        values = self.critic(states).squeeze().detach().numpy()

        # 计算advantages
        advantages = self.compute_gae(rewards, values, dones)
        advantages = torch.FloatTensor(advantages)

        # 归一化advantages
        advantages = (advantages - advantages.mean()) / (advantages.std() + 1e-8)

        # PPO更新
        states = torch.FloatTensor(states)
        actions = torch.FloatTensor(actions)
        old_log_probs = torch.FloatTensor(old_log_probs)

        for _ in range(self.clip_epochs):
            # 重新采样（简化版）
            indices = torch.randint(0, len(states), (self.batch_size,))

            batch_states = states[indices]
            batch_actions = actions[indices]
            batch_old_log_probs = old_log_probs[indices]
            batch_advantages = advantages[indices]

            # 计算新的log_prob
            mu, std = self.actor(batch_states)
            dist = Normal(mu, std)
            new_log_probs = dist.log_prob(batch_actions).sum(dim=-1)

            # 计算ratio
            ratio = torch.exp(new_log_probs - batch_old_log_probs)

            # PPO损失
            surr1 = ratio * batch_advantages
            surr2 = torch.clamp(ratio, 1 - self.epsilon, 1 + self.epsilon) * batch_advantages
            actor_loss = -torch.min(surr1, surr2).mean()

            # 更新Actor
            self.actor_optimizer.zero_grad()
            actor_loss.backward()
            self.actor_optimizer.step()

            # 更新Critic
            values_pred = self.critic(batch_states).squeeze()
            returns = batch_advantages + values_pred.detach()
            critic_loss = nn.MSELoss()(values_pred, returns)

            self.critic_optimizer.zero_grad()
            critic_loss.backward()
            self.critic_optimizer.step()

# 应用：充电电流控制
def charging_current_control():
    """
    充电电流控制（连续动作空间）

    状态：[SOC, 温度, 电压, 内阻]
    动作：充电电流（连续值，0-2C）
    """
    state_dim = 4
    action_dim = 1  # 充电电流

            state = next_state

        # 更新策略
        agent.update(states, actions, log_probs, rewards, dones)

    return agent
```
