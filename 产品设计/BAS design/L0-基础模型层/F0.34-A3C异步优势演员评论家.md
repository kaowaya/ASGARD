# F0.34 A3C异步优势演员评论家

## 算法概述

A3C是一种异步并行的Actor-Critic算法，通过多个agent并行探索提高样本效率。在电池管理系统中，A3C适用于：

- **大规模并行训练**：加速训练
- **复杂控制任务**：连续控制
- **多场景学习**：不同工况

## 核心思想

**多线程异步训练**：
- 每个线程独立探索
- 异步更新全局网络
- 提高样本多样性

## 代码示例

```python
import torch
import torch.nn as nn
import torch.multiprocessing as mp
import torch.optim as optim

class ActorCritic(nn.Module):
    """
    Actor-Critic网络
    """
    def __init__(self, state_dim, action_dim):
        super().__init__()

        # 共享特征提取
        self.features = nn.Sequential(
            nn.Linear(state_dim, 64),
            nn.Tanh()
        )

        # Actor（策略）
        self.actor = nn.Sequential(
            nn.Linear(64, action_dim),
            nn.Softmax(dim=-1)
        )

        # Critic（价值）
        self.critic = nn.Linear(64, 1)

    def forward(self, x):
        x = self.features(x)
        policy = self.actor(x)
        value = self.critic(x)
        return policy, value

class A3CAgent:
    """
    A3C智能体
    """
    def __init__(self, state_dim, action_dim, lr=1e-4):
        self.state_dim = state_dim
        self.action_dim = action_dim

        # 全局网络
        self.global_model = ActorCritic(state_dim, action_dim)
        self.global_model.share_memory()

        # 优化器
        self.optimizer = optim.Adam(self.global_model.parameters(), lr=lr)

        # 超参数
        self.gamma = 0.99
        self.t_max = 20  # 最大步数

    def worker(self, worker_id, env):
        """
        工作线程
        """
        # 本地网络
        local_model = ActorCritic(self.state_dim, self.action_dim)
        local_model.load_state_dict(self.global_model.state_dict())

        state = env.reset()
        done = False

        # 存储经验
        states, actions, rewards = [], [], []

        for t in range(self.t_max):
            # 选择动作
            state_tensor = torch.FloatTensor(state).unsqueeze(0)
            policy, value = local_model(state_tensor)

            # 采样动作
            dist = torch.distributions.Categorical(policy)
            action = dist.sample().item()

            # 执行动作
            next_state, reward, done = env.step(action)

            # 存储经验
            states.append(state)
            actions.append(action)
            rewards.append(reward)

            if done:
                break

            state = next_state

        # 计算R（bootstrap）
        R = 0
        if not done:
            state_tensor = torch.FloatTensor(state).unsqueeze(0)
            _, value = local_model(state_tensor)
            R = value.item()

        # 计算returns
        returns = []
        for r in reversed(rewards):
            R = r + self.gamma * R
            returns.insert(0, R)

        # 转为tensor
        states = torch.FloatTensor(states)
        actions = torch.LongTensor(actions)
        returns = torch.FloatTensor(returns)

        # 计算loss
        policy, values = local_model(states)
        values = values.squeeze()

        # Actor loss（策略梯度）
        dist = torch.distributions.Categorical(policy)
        log_probs = dist.log_prob(actions)
        advantages = returns - values.detach()
        actor_loss = -(log_probs * advantages).mean()

        # Critic loss（价值估计）
        critic_loss = nn.MSELoss()(values, returns)

        # 总loss
        loss = actor_loss + 0.5 * critic_loss - 0.01 * dist.entropy().mean()

        # 更新全局网络
        self.optimizer.zero_grad()
        loss.backward()
        for global_param, local_param in zip(self.global_model.parameters(), local_model.parameters()):
            global_param._grad = local_param.grad
        self.optimizer.step()

        # 同步本地网络
        local_model.load_state_dict(self.global_model.state_dict())

    def train(self, envs, n_workers=4, n_episodes=1000):
        """
        多线程训练
        """
        processes = []

        for episode in range(n_episodes):
            # 启动工作线程
            for worker_id in range(n_workers):
                env = envs[worker_id]
                p = mp.Process(target=self.worker, args=(worker_id, env))
                p.start()
                processes.append(p)

            # 等待所有线程完成
            for p in processes:
                p.join()

            processes = []

            if episode % 10 == 0:
                print(f"Episode {episode}")

        return self.global_model

# 应用：并行训练
```
