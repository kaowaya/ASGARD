# F0.33 GRPO群体相对策略优化

## 算法概述

GRPO是一种策略梯度算法，通过群体相对优势估计，去除基线。在电池管理系统中，GRPO主要用于：

- **充电策略优化**：多目标充电优化
- **V2G调度**：车辆到电网调度
- **能量管理**：储能系统能量管理

## GRPO核心思想

```
1. 收集群体 trajectories
2. 计算群体平均 reward
3. 去基线: advantage = reward - baseline
4. 策略更新: ∇J ≈ Σ(advantage · ∇log π(a|s))
```

## 代码示例

```python
import torch
import torch.nn as nn
import torch.optim as optim
from torch.distributions import Normal

class GRPOAgent:
    """
    GRPO智能体
    """
    def __init__(self, state_dim, action_dim, lr=1e-4, group_size=10):
        # 策略网络
        self.policy = nn.Sequential(
            nn.Linear(state_dim, 64),
            nn.Tanh(),
            nn.Linear(64, 64),
            nn.Tanh(),
            nn.Linear(64, action_dim)
        )

        self.optimizer = optim.Adam(self.policy.parameters(), lr=lr)

        self.group_size = group_size
        self.gamma = 0.99

    def select_action(self, state):
        """
        选择动作
        """
        with torch.no_grad():
            state = torch.FloatTensor(state).unsqueeze(0)
            action = self.policy(state)
            return action[0].numpy()

    def collect_trajectories(self, env, n_trajectories):
        """
        收集群体trajectories
        """
        trajectories = []

        for _ in range(n_trajectories):
            state = env.reset()
            trajectory = []

            done = False
            while not done:
                action = self.select_action(state)
                next_state, reward, done = env.step(action)

                trajectory.append({
                    'state': state,
                    'action': action,
                    'reward': reward
                })

                state = next_state

            trajectories.append(trajectory)

        return trajectories

    def compute_group_returns(self, trajectories):
        """
        计算群体return（去基线）
        """
        returns = []

        for traj in trajectories:
            # 计算discounted return
            R = 0
            discounted_rewards = []

            for step in reversed(traj):
                R = step['reward'] + self.gamma * R
                discounted_rewards.insert(0, R)

            returns.append(sum(discounted_rewards))

        # 计算基线（群体平均）
        baseline = sum(returns) / len(returns)

        # 去基线
        advantages = [R - baseline for R in returns]

        return advantages

    def update(self, trajectories):
        """
        更新策略
        """
        # 计算advantages
        advantages = self.compute_group_returns(trajectories)

        # 归一化advantages
        advantages = torch.FloatTensor(advantages)
        advantages = (advantages - advantages.mean()) / (advantages.std() + 1e-8)

        # 策略更新
        for traj, adv in zip(trajectories, advantages):
            for step in traj:
                state = torch.FloatTensor(step['state'])
                action = torch.FloatTensor(step['action'])

                # 前向传播
                action_pred = self.policy(state.unsqueeze(0))

                # 计算损失（简化版：MSE）
                loss = (action_pred - action).pow(2).mean()

                # 加权advantage
                weighted_loss = adv * loss

                # 更新
                self.optimizer.zero_grad()
                weighted_loss.backward()
                self.optimizer.step()

# 应用：充电策略优化
def charging_strategy_optimization():
    """
    充电策略优化（多目标）
    """
    state_dim = 4  # SOC, T, V, I
    action_dim = 1  # 充电电流

    agent = GRPOAgent(state_dim, action_dim, group_size=10)

    # 训练循环
    for iteration in range(100):
        # 收集群体trajectories
        trajectories = agent.collect_trajectories(env, agent.group_size)

        # 更新策略
        agent.update(trajectories)

    return agent
```
