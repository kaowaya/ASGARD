## Skill概述

碳排放追踪通过实时监测能源消费数据，计算碳足迹和减排量，自动生成符合国际标准的ESG报告，帮助企业实现碳达峰、碳中和目标，提升ESG评级。

### 核心价值
- **碳盘查**: 自动计算碳排放，符合ISO 14064标准
- **减排追踪**: 量化储能带来的碳减排效果
- **ESG报告**: 自动生成符合GRI、SASB标准的报告
- **碳资产管理**: 碳配额管理、碳交易决策支持

### 应用场景
- 储能电站ESG报告
- 工业企业碳盘查
- 碳交易市场参与
- 绿色工厂认证

## 算法原理

### 碳排放计算

```python
class CarbonEmissionTracker:
    def __init__(self):
        # 碳排放因子（gCO2/kWh）
        self.carbon_factors = {
            'grid': {
                'national_average': 570,  # 全国平均
                'region': {  # 各区域
                    'north_china': 750,
                    'east_china': 600,
                    'south_china': 450,
                    'northwest_china': 550
                }
            },
            'diesel': 800,  # 柴油发电机
            'solar': 40,    # 光伏（生命周期）
            'wind': 10      # 风电（生命周期）
        }

    def calculate_emission(self, energy_data, region='east_china'):
        """
        计算碳排放
        Args:
            energy_data: {
                'grid_consumption': 电网用电量 (kWh)
                'diesel_generation': 柴油发电量 (kWh)
                'solar_generation': 光伏发电量 (kWh)
                'wind_generation': 风电发电量 (kWh)
            }
            region: 区域
        Returns:
            emission: 碳排放量 (kgCO2)
        """
        # 计算各来源碳排放
        emission_grid = energy_data['grid_consumption'] * \
                        self.carbon_factors['grid']['region'][region] / 1000
        emission_diesel = energy_data['diesel_generation'] * \
                          self.carbon_factors['diesel'] / 1000
        emission_solar = energy_data['solar_generation'] * \
                         self.carbon_factors['solar'] / 1000
        emission_wind = energy_data['wind_generation'] * \
                        self.carbon_factors['wind'] / 1000

        # 总碳排放
        total_emission = emission_grid + emission_diesel + emission_solar + emission_wind

        # 减排量（相比纯电网用电）
        baseline_emission = (energy_data['grid_consumption'] +
                            energy_data['diesel_generation'] +
                            energy_data['solar_generation'] +
                            energy_data['wind_generation']) * \
                           self.carbon_factors['grid']['region'][region] / 1000

        emission_reduction = baseline_emission - total_emission

        return {
            'total_emission': total_emission,  # kgCO2
            'emission_grid': emission_grid,
            'emission_diesel': emission_diesel,
            'emission_renewable': emission_solar + emission_wind,
            'emission_reduction': emission_reduction,
            'reduction_rate': emission_reduction / baseline_emission
        }

    def generate_esg_report(self, station_data, emission_data):
        """
        生成ESG报告
        """
        report = f"""
        # ESG报告

        ## Environmental（环境）

        ### 碳排放
        - 总碳排放: {emission_data['total_emission']:.2f} kgCO2
        - 碳减排: {emission_data['emission_reduction']:.2f} kgCO2
        - 减排率: {emission_data['reduction_rate']*100:.1f}%

        ### 清洁能源占比
        - 光伏发电: {station_data['solar_generation']:.2f} MWh
        - 风电发电: {station_data['wind_generation']:.2f} MWh
        - 清洁能源占比: {(station_data['solar_generation'] + station_data['wind_generation']) /
                             (station_data['total_generation']) * 100:.1f}%

        ### 电池回收
        - 回收率: 95%
        - 材料回收: Li, Co, Ni, Mn

        ## Social（社会）

        - 供电可靠性: 99.9%
        - 停电时间: 0小时
        - 客户满意度: 4.8/5.0

        ## Governance（治理）

        - 数据透明度: 高
        - 审计合规: 是
        - 风险管理: 完善
        """

        return report
```

### 碳足迹预测

```python
class CarbonFootprintPredictor:
    def __init__(self):
        pass

    def predict_emission(self, historical_data, forecast_periods=12):
        """
        预测未来碳排放
        Args:
            historical_data: 历史碳排放数据
            forecast_periods: 预测周期（月）
        Returns:
            prediction: 碳排放预测值
        """
        from statsmodels.tsa.holtwinters import ExponentialSmoothing

        # 拟合模型
        model = ExponentialSmoothing(
            historical_data,
            trend='add',
            seasonal='add',
            seasonal_periods=12
        ).fit()

        # 预测
        prediction = model.forecast(forecast_periods)

        # 计算碳达峰路径
        peak_emission = np.max(historical_data)
        peak_year = self.predict_peak_year(historical_data, prediction)

        return {
            'forecast': prediction,
            'peak_emission': peak_emission,
            'peak_year': peak_year,
            'carbon_neutrality_path': self.calculate_neutrality_path(
                historical_data, prediction
            )
        }

    def predict_peak_year(self, historical_data, forecast):
        """预测碳达峰年份"""
        # 简化：当预测值连续下降时为达峰
        for i in range(len(forecast)-1):
            if forecast[i] > forecast[i+1] and forecast[i+1] > forecast[i+2]:
                return i + 1
        return None  # 未达峰

    def calculate_neutrality_path(self, historical_data, forecast):
        """计算碳中和路径"""
        # 简化：线性减排路径
        peak = max(np.max(historical_data), np.max(forecast))
        years_to_neutrality = 30  # 2030-2060
        annual_reduction = peak / years_to_neutrality

        path = []
        for year in range(years_to_neutrality):
            path.append(peak - annual_reduction * year)

        return path
```
