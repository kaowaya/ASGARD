### DRL模型设计

```python
import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim

class V2GAgent:
    def __init__(self, state_dim, action_dim):
        """
        V2G智能体
        Args:
            state_dim: 状态维度 [SOC, 电池健康, 时间, 电价, 需求]
            action_dim: 动作维度 [充电功率, 放电功率, 待机]
        """
        self.state_dim = state_dim
        self.action_dim = action_dim

        # Actor网络（策略网络）
        self.actor = nn.Sequential(
            nn.Linear(state_dim, 128),
            nn.ReLU(),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Linear(64, action_dim),
            nn.Softmax(dim=-1)
        )

        # Critic网络（价值网络）
        self.critic = nn.Sequential(
            nn.Linear(state_dim, 128),
            nn.ReLU(),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Linear(64, 1)
        )

        self.optimizer_actor = optim.Adam(self.actor.parameters(), lr=1e-4)
        self.optimizer_critic = optim.Adam(self.critic.parameters(), lr=1e-3)

    def select_action(self, state):
        """
        选择动作
        Args:
            state: 当前状态
        Returns:
            action: 动作
        """
        state = torch.FloatTensor(state)
        action_probs = self.actor(state)
        action_dist = torch.distributions.Categorical(action_probs)
        action = action_dist.sample()
        return action.item()

    def train(self, replay_buffer, batch_size=64):
        """
        训练（PPO算法）
        Args:
            replay_buffer: 经验回放缓冲区
            batch_size: 批次大小
        """
        # 采样批次
        states, actions, rewards, next_states, dones = replay_buffer.sample(batch_size)

        states = torch.FloatTensor(states)
        actions = torch.LongTensor(actions)
        rewards = torch.FloatTensor(rewards)
        next_states = torch.FloatTensor(next_states)
        dones = torch.FloatTensor(dones)

        # 计算TD目标
        with torch.no_grad():
            td_target = rewards + 0.99 * (1 - dones) * self.critic(next_states)

        # 更新Critic
        value_pred = self.critic(states).squeeze()
        critic_loss = nn.MSELoss()(value_pred, td_target)
        self.optimizer_critic.zero_grad()
        critic_loss.backward()
        self.optimizer_critic.step()

        # 更新Actor（PPO）
        action_probs = self.actor(states)
        action_dist = torch.distributions.Categorical(action_probs)
        log_probs = action_dist.log_prob(actions)
        entropy = action_dist.entropy().mean()

        # PPO损失
        ratio = torch.exp(log_probs - log_probs.detach())  # 重要性采样比率
        advantages = td_target - value_pred.detach()
        surrogate_loss = ratio * advantages

        actor_loss = -surrogate_loss.mean() + 0.01 * entropy  # 加入熵正则化
        self.optimizer_actor.zero_grad()
        actor_loss.backward()
        torch.nn.utils.clip_grad_norm_(self.actor.parameters(), 0.5)
        self.optimizer_actor.step()

        return actor_loss.item(), critic_loss.item()

    def calculate_reward(self, state, action, next_state):
        """
        计算奖励
        考虑因素:
          1. 经济收益
          2. 电池寿命损耗
          3. 用户满意度（SOC满足出行需求）
        """
        soc_current, soh, time, price, demand = state
        soc_next, soh_next, time_next, price_next, demand_next = next_state

        # 1. 经济收益
        revenue = 0
        if action == 1:  # 放电
            revenue = price * min(20, soc_current * 50) * 0.9  # 假设20kW放电，90%效率

        # 2. 电池损耗
        battery_degradation = 0
        if action == 1 or action == 0:  # 放电或充电
            # 简化：每次充放电造成0.001% SOH损失
            battery_degradation = (soh - soh_next) * 1000  # 转换为成本

        # 3. 用户满意度
        satisfaction = 0
        if soc_next < 0.2:  # SOC过低
            satisfaction = -10

        # 总奖励
        reward = revenue - battery_degradation + satisfaction

        return reward
```

        self.num_evs = num_evs
        self.evs = [EV(i) for i in range(num_evs)]

    def optimize_v2g_schedule(self, grid_demand, price_forecast):
        """
        优化V2G调度
        """
        schedule = {}

        for t in range(24):
            # 收集所有EV状态
            states = []
            for ev in self.evs:
                state = [
                    ev.soc,
                    ev.soh,
                    t / 24,  # 时间归一化
                    price_forecast[t] / 2,  # 电价归一化
                    grid_demand[t] / 1000  # 需求归一化
                ]
                states.append(state)

            # 为每个EV选择动作
            actions = []
            for state in states:
                action = self.agent.select_action(state)
                actions.append(action)

            # 执行动作
            for i, ev in enumerate(self.evs):
                next_state, reward = ev.execute_action(actions[i], price_forecast[t])

                # 存储经验
                self.replay_buffer.add(states[i], actions[i], reward, next_state, False)

            # 训练智能体
            if len(self.replay_buffer) > 1000:
                actor_loss, critic_loss = self.agent.train(self.replay_buffer)

            # 统计调度结果
            schedule[t] = {
                'total_discharge': sum([self.evs[i].power for i in range(self.num_evs) if actions[i] == 1]),
                'total_charge': sum([self.evs[i].power for i in range(self.num_evs) if actions[i] == 0]),
                'idle': sum([1 for i in range(self.num_evs) if actions[i] == 2])
            }

        return schedule

## 技术特点

### 1. 深度强化学习（DRL）
- **PPO算法**: 近端策略优化，稳定高效
- **Actor-Critic架构**: 策略网络+价值网络
- **经验回放**: 提升样本利用效率
- **奖励函数设计**: 平衡经济收益、电池寿命和用户满意度

### 2. 多智能体协同
- **分布式决策**: 每辆EV独立决策
- **聚合协调**: 聚合商统一调度
- **需求响应**: 快速响应电网需求信号

### 3. 约束保障
- **SOC约束**: 保留20%电量满足出行需求
- **SOH保护**: 避免深度充放电，延长电池寿命
- **时间窗口**: 在用户设定的时间窗口内充电
