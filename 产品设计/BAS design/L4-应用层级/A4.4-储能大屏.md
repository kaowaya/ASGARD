## 代码示例

### React组件

```jsx
import React, { useState, useEffect } from 'react';
import ReactECharts from 'echarts-for-react';
import { WebSocket } from 'websocket';

const StorageDashboard = () => {
  const [data, setData] = useState({
    totalPower: 0,
    totalCapacity: 0,
    todayRevenue: 0,
    socDistribution: [],
    powerCurve: [],
    alerts: [],
    deviceStatus: []
  });

  useEffect(() => {
    // WebSocket连接
    const ws = new WebSocket('ws://localhost:8080/api/realtime');

    ws.onmessage = (message) => {
      const update = JSON.parse(message.data);
      setData(update);
    };

    return () => ws.close();
  }, []);

  // SOC热力图配置
  const socHeatmapOption = {
    title: { text: 'SOC分布热力图' },
    xAxis: { type: 'category', data: data.socDistribution.map(d => d.rack) },
    yAxis: { type: 'category', data: data.socDistribution.map(d => d.module) },
    visualMap: { min: 0, max: 100, inRange: { color: ['#50a3ba', '#eac736', '#d94e5d'] } },
    series: [{
      type: 'heatmap',
      data: data.socDistribution.map(d => [d.rack, d.module, d.soc])
    }]
  };

  // 功率曲线配置
  const powerCurveOption = {
    title: { text: '实时功率曲线' },
    xAxis: { type: 'category', data: data.powerCurve.map(d => d.time) },
    yAxis: { type: 'value', name: '功率(MW)' },
    series: [
      { name: '光伏', type: 'line', data: data.powerCurve.map(d => d.pv) },
      { name: '负载', type: 'line', data: data.powerCurve.map(d => d.load) },
      { name: '电池', type: 'line', data: data.powerCurve.map(d => d.battery) },
      { name: '电网', type: 'line', data: data.powerCurve.map(d => d.grid) }
    ]
  };

  return (
    <div className="dashboard">
      <div className="header">
        <div className="card">
          <h3>总功率</h3>
          <p>{data.totalPower} MW</p>
        </div>
        <div className="card">
          <h3>总容量</h3>
          <p>{data.totalCapacity} MWh</p>
        </div>
        <div className="card">
          <h3>今日收益</h3>
          <p>¥{data.todayRevenue}</p>
        </div>
      </div>

      <div className="charts">
        <ReactECharts option={socHeatmapOption} />
        <ReactECharts option={powerCurveOption} />
      </div>

      <div className="sidebar">
        <div className="alerts">
          <h3>告警列表</h3>
          <ul>
            {data.alerts.map((alert, i) => (
              <li key={i} className={alert.level}>{alert.message}</li>
            ))}
          </ul>
        </div>

        <div className="status">
          <h3>设备状态</h3>
          {data.deviceStatus.map((device, i) => (
            <div key={i} className={device.status}>
              {device.name}: {device.status}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
};

export default StorageDashboard;
```

## 技术特点

### 优势

1. **实时性**: WebSocket实时推送数据，延迟<100ms
2. **可视化**: 图表化呈现，直观易懂
3. **响应式**: 适配不同屏幕尺寸
4. **100%通用**: 适用于所有类型的储能系统

### 局限性

1. **网络依赖**: 需要稳定的网络连接
2. **浏览器性能**: 大量数据渲染可能影响性能
3. **数据安全**: 需要考虑数据安全传输
4. **定制化**: 不同场景需要定制开发
